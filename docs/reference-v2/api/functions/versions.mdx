---
title: "List Function Versions"
api: "GET /v2/functions/{organization_name}/{function_name}/versions"
---

This endpoint returns a list of function versions of a function. 

### Response

<ResponseField name="id" type="string">
  Id of the model.
</ResponseField>
<ResponseField name="build_status" type="string">
  Build status of the function version. Statuses are 'queued', 'building', 'pushing', 'deploying', 'ready', 'error'
</ResponseField>
<ResponseField name="queued_at" type="string">
  The time that the model build is queued.
</ResponseField>
<ResponseField name="built_at" type="string">
  The time that the model build finished building.
</ResponseField>
<ResponseField name="ready_at" type="string">
  The time that the model is ready to process jobs.
</ResponseField>
<ResponseField name="compute_type" type="string">
  Compute Type (TODO: Finalize the types)
</ResponseField>
<ResponseField name="python_packages" type="string array">
  An array of the model's python pip packages.
</ResponseField>
<ResponseField name="system_packages" type="string array">
  An array of the model's system packages.
</ResponseField>
<ResponseField name="cuda_version" type="string">
  Cuda Version.
</ResponseField>
<ResponseField name="gpu" type="boolean">
  Whether this model uses gpu.
</ResponseField>
<ResponseField name="minimum_replicas" type="integer">
    The minimum number of replicas of the model. Scale your model's minimum replicas to avoid cold starts.
</ResponseField>
<ResponseField name="maximum_replicas" type="integer">
    The maximum number of replicas of the model. Prevents model replicas from scaling beyond a certain point.
</ResponseField>
<ResponseField name="inputs" type="array Input">
The inputs to a function
    <Expandable title="properties">
        <ResponseField name="type" type="string">
        TODO: Copy paste later
        </ResponseField>
        <ResponseField name="name" type="string">
        The name of this input.
        </ResponseField>
        <ResponseField name="data" type="string">
        The data of the input. Empty if the response if not from a job. 
        </ResponseField>
        <ResponseField name="description" type="string">
        The description of the input.
        </ResponseField>
    </Expandable>
</ResponseField>
<ResponseField name="outputs" type="array Output">
The outputs to a function
    <Expandable title="properties">
        <ResponseField name="type" type="string">
        TODO: Copy paste later
        </ResponseField>
        <ResponseField name="name" type="string">
        The name of this input.
        </ResponseField>
        <ResponseField name="data" type="string">
        The data of the input. Empty if the response if not from a job. 
        </ResponseField>
        <ResponseField name="description" type="string">
        The description of the input.
        </ResponseField>
    </Expandable>
</ResponseField>
<ResponseField name="environment_variables" type="array EnvironmentVariable">
The outputs to a function
    <Expandable title="properties">
        <ResponseField name="name" type="string">
        The name of the environment variables
        </ResponseField>
        <ResponseField name="description" type="string">
        The description of the environment variables
        </ResponseField>
        <ResponseField name="default" type="string">
        Default value of the environment variable
        </ResponseField>
    </Expandable>
</ResponseField>
