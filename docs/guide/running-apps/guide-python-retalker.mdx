```python
# TODO: Use public links
def retalker(): 
    source_video=sieve.Video(
            url="https://storage.googleapis.com/sieve-prod-us-central1-public-file-upload-bucket/c4d968f5-f25a-412b-9102-5b6ab6dafcb4/video_retalker/df431771-8c96-41aa-8c86-5f0179ef3f19-input-source_video.mp4")
    target_audio=sieve.Audio(
            url="https://storage.googleapis.com/sieve-prod-us-central1-public-file-upload-bucket/c4d968f5-f25a-412b-9102-5b6ab6dafcb4/video_retalker/df431771-8c96-41aa-8c86-5f0179ef3f19-input-target_audio.wav")
    retalker = sieve.function.get("sieve-internal/video_retalker")
    sieve_video = retalker.run(source_video, target_audio)
    return sieve_video.url

video_url = retalker()

print(video_url)
```

Run the code and view the processed video in the outputted url

