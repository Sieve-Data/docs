```python BoundingBox
class BoundingBox(SieveBaseModel, validate_assignment=True):
    """
    A bounding box is a 2D box in space that denotes the location of an object
    """
    x1: float
    y1: float
    x2: float
    y2: float

    def get_list(self):
        """
        Returns the bounding box as a list
        """

    def get_array(self):
        """
        Returns the bounding box as a numpy array
        """

    def from_array(array: np.ndarray):
        """
        Creates a bounding box from a numpy array
        """

    def center(self) -> Point:
        """
        Returns the center of the bounding box
        """
```
